\begin{align*}
F&=\overline{S1}(\overline{S0})I0+\overline{S1}S0I1+S1\overline{S0}I2+S1S0I3\\ 
F&=\overline{P}(\overline{Q})(R)+\overline{P}Q(0)+P\overline{Q}(R)+PQ(1)\\
F&={P}{Q}R+P{Q}R+PQ
\end{align*}
